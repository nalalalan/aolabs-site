\roleentry{Worcester Polytechnic Institute, Worcester, MA}{Aug 2023 -- Present}{Graduate Researcher -- NanoEnergy Lab and Soft Robotics Lab}{Advisors: Dr. Pratap Rao and Dr. Cagdas Onal}
\begin{itemize}
  \item Lead a first-author manuscript on a reconfigurable cellular soft-robotics architecture that uses one monolithically printed pneumatic Sarrus/PneuNet unit to create multiple robotic topologies.
  \item Designed, fabricated, and characterized 2-by-2 pneumatic modules with integrated cross-cell channels, snap-fit connectivity, and a built-in bending bias for repeatable out-of-plane morphing.
  \item Built planar arrays for programmable morphing surfaces, demonstrating single- and double-dome shape display, bilayer overhangs, tennis-ball transport/capture, and horizontal, diagonal, and circular traveling-wave deformation.
  \item Reconfigured the same modules into cylindrical soft robots for C-curve and S-curve continuum bending, object grasping and placement, peristaltic crawling of approximately 200 mm, and self-rolling locomotion.
  \item Measured pressure-dependent expansion, bending angle, force-displacement response, hysteresis, and variable stiffness across cell, module, and array scales to connect geometry, actuation pattern, and task-level behavior.
\end{itemize}

\roleentry{Sandia National Laboratories, Albuquerque, NM}{June 2025 -- Aug 2025}{Graduate Research Intern -- Nonlinear Mechanics and Dynamics (NOMAD) Research Institute}{}
\begin{itemize}
  \item Co-developed a NOMAD parameter study quantifying how nonphysical finite-element settings affect nonlinear dynamics predictions for an electromechanical ratcheting mechanism with intermittent contact, friction, springs, and pawl-gear interactions.
  \item Built and compared idealized analytical, MATLAB multibody, and finite-element submodels for pin-spring-pawl, pin-pawl, and pawl-gear interactions under haversine shock and sinusoidal vibration.
  \item Ran structured sweeps across momentum-balance iterations (1--100), mesh density (fine to very coarse), and processor count (50--250\%) to separate physical response from solver and modeling sensitivity.
  \item Identified distinct sensitivity regimes: pin-spring-pawl converged across tested parameters, while pin-pawl and pawl-gear cases showed contact/friction-driven divergence, mesh-related lock-up, and environment-dependent sensitivity.
  \item Presented final results showing that simulation sensitivity appears at submodel scale, informing more defensible setup choices before extending studies to full assembly-level ratcheting mechanisms.
\end{itemize}
